	SELECT*FROM produit WHERE produit.nom LIKE "C%";
2. Récupérer la liste de tous les produits contenant un "C"
	SELECT*FROM produit WHERE produit.nom LIKE "%C%";
3. Récupérer tous les clients habitant à Rennes
	SELECT DISTINCT*FROM client, adresse WHERE client.id_adresse = adresse.id AND adresse.ville = "Rennes";
4. Récupérer tous les paniers d'un client
	SELECT client.nom,client.id AS client_id, panier.id as panier_id FROM panier,client WHERE panier.id_client = client.id AND client.id = 1;
5. Récupérer le nombre de produits de chaque panier
	SELECT contient.id_panier, COUNT(*) AS nombre_produit FROM contient GROUP BY contient.id_panier;
6. Calculez le prix du panier d'un client
	SELECT contient.id_panier, SUM(produit.prix) AS prix_panier FROM contient, produit WHERE contient.id_produit = produit.id GROUP by contient.id_panier;
7. Récupérer tous les paniers contenant le produit : ( un des produits que vous avez créés)
	SELECT panier.* FROM contient,panier WHERE contient.id_produit = 1 AND contient.id_produit = panier.id;
8. Récupérer tous les clients ayant un panier
	SELECT DISTINCT client.* FROM client, panier WHERE client.id = panier.id;
